Aim 2: SILA Analysis

Overview

This analysis uses SILA to estimate time to amyloid positivity threshold from longitudinal plasma biomarker data.

Biomarkers analyzed:

  • p-tau181
  • p-tau217

Amyloid positivity thresholds:

  • Centiloid ≥ 22.32 (PREVENT-AD cohort-specific)

Biomarker cutoffs: Derived from ROC analysis (Youden index optimal cutoffs) in PREVENT-AD_PET.qmd - p-tau181: 6.42 pg/mL - p-tau217: 3.06 pg/mL

# Function to find closest biomarker measurement to PET scan
find_closest_biomarker <- function(pet_data, biomarker_data, biomarker_col) {

  # Convert dates to Date objects
  pet_data <- pet_data %>%
    mutate(pet_date_parsed = ymd(paste0(pet_date, "-01")))

  biomarker_data <- biomarker_data %>%
    mutate(biomarker_date_parsed = ymd(paste0(Date_taken, "-01")))

  # For each PET scan, find closest biomarker
  matched_data <- pet_data %>%
    left_join(biomarker_data, by = "CONP_ID", relationship = "many-to-many") %>%
    mutate(
      date_diff_months = abs(interval(biomarker_date_parsed, pet_date_parsed) / months(1))
    ) %>%
    group_by(CONP_ID, pet_session) %>%
    slice_min(date_diff_months, n = 1, with_ties = FALSE) %>%
    ungroup()

  return(matched_data)
}

# Function to run SILA for a single biomarker
run_sila_biomarker <- function(data, biomarker_col, threshold_val, biomarker_name) {

  cat("\n--- Running SILA for", biomarker_name, "---\n")
  cat("Threshold:", threshold_val, "\n")

  # Prepare data in SILA format
  sila_data <- data %>%
    filter(!is.na(.data[[biomarker_col]])) %>%
    filter(!is.na(age)) %>%
    filter(is.finite(.data[[biomarker_col]])) %>%
    filter(is.finite(age)) %>%
    select(subid = CONP_ID, age, val = all_of(biomarker_col)) %>%
    arrange(subid, age)

  # Remove extreme outliers (beyond 3 SD from mean, or use IQR method)
  q1 <- quantile(sila_data$val, 0.25, na.rm = TRUE)
  q3 <- quantile(sila_data$val, 0.75, na.rm = TRUE)
  iqr <- q3 - q1
  lower_bound <- q1 - 3 * iqr
  upper_bound <- q3 + 3 * iqr

  n_before <- nrow(sila_data)
  sila_data <- sila_data %>%
    filter(val >= lower_bound & val <= upper_bound)
  n_after <- nrow(sila_data)

  if (n_before != n_after) {
    cat("Removed", n_before - n_after, "extreme outliers\n")
  }

  # Check minimum requirements
  n_subjects <- n_distinct(sila_data$subid)
  n_obs <- nrow(sila_data)

  cat("N subjects:", n_subjects, "\n")
  cat("N observations:", n_obs, "\n")
  cat("Value range:", round(min(sila_data$val), 2), "-", round(max(sila_data$val), 2), "\n")

  # Check for any remaining issues
  if (any(!is.finite(sila_data$val)) || any(!is.finite(sila_data$age))) {
    cat("ERROR: Non-finite values detected after cleaning\n")
    return(NULL)
  }

  if (n_subjects < 10) {
    warning(paste("Insufficient subjects for", biomarker_name, ":", n_subjects))
    return(NULL)
  }

  # Ensure threshold is within data range
  if (threshold_val < min(sila_data$val) || threshold_val > max(sila_data$val)) {
    cat("WARNING: Threshold", threshold_val, "is outside data range\n")
    cat("Adjusting to median value:", round(median(sila_data$val), 2), "\n")
    threshold_val <- median(sila_data$val)
  }

  # Run SILA
  tryCatch({
    cat("Running sila()...\n")
    res <- sila(sila_data, dt = 0.25, val0 = threshold_val, maxi = 200)
    cat("Running sila_estimate()...\n")
    resfit <- sila_estimate(res$tsila, sila_data)
    cat("SILA completed successfully\n")

    list(
      biomarker = biomarker_name,
      threshold = threshold_val,
      sila_result = res,
      sila_fit = resfit,
      n_subjects = n_subjects,
      n_obs = n_obs
    )
  }, error = function(e) {
    cat("ERROR:", e$message, "\n")
    warning(paste("SILA failed for", biomarker_name, ":", e$message))
    return(NULL)
  })
}

# Function to create SILA plots
plot_sila_results <- function(sila_output, title_suffix = "", amyloid_lookup = NULL) {

  if (is.null(sila_output)) return(NULL)

  res <- sila_output$sila_result
  resfit <- sila_output$sila_fit
  biomarker <- sila_output$biomarker
  threshold <- sila_output$threshold

  # Join amyloid status if provided
  if (!is.null(amyloid_lookup)) {
    resfit <- resfit %>%
      left_join(amyloid_lookup, by = c("subid" = "CONP_ID")) %>%
      mutate(amyloid_status = ifelse(is.na(amyloid_status), "Unknown", as.character(amyloid_status)))
  }

  # DRS curve
  p1 <- ggplot(res$tdrs, aes(x = val, y = rate)) +
    geom_ribbon(aes(ymin = rate - ci, ymax = rate + ci), alpha = 0.3, fill = "#2166AC") +
    geom_line(color = "#2166AC", linewidth = 1) +
    geom_vline(xintercept = threshold, linetype = "dashed", color = "red") +
    labs(
      title = paste(biomarker, "- Discrete Rate Sampling"),
      subtitle = title_suffix,
      x = biomarker,
      y = expression(Delta * " per Year")
    ) +
    theme_minimal()

  # SILA modeled curve
  p2 <- ggplot(res$tsila, aes(x = adtime, y = val)) +
    geom_line(color = "#2166AC", linewidth = 1) +
    geom_hline(yintercept = threshold, linetype = "dashed", color = "red") +
    labs(
      title = paste(biomarker, "- SILA Modeled Curve"),
      subtitle = title_suffix,
      x = "Time from Threshold (years)",
      y = biomarker
    ) +
    theme_minimal()

  # Individual trajectories aligned to threshold - colored by amyloid status
  if (!is.null(amyloid_lookup) && "amyloid_status" %in% names(resfit)) {
    p3 <- ggplot(resfit, aes(x = estdtt0, y = val, group = subid, color = amyloid_status)) +
      geom_line(alpha = 0.4) +
      geom_point(alpha = 0.5, size = 0.8) +
      geom_line(data = res$tsila, aes(x = adtime, y = val, group = NULL, color = NULL),
                color = "black", linewidth = 1.2) +
      geom_hline(yintercept = threshold, linetype = "dashed", color = "red") +
      scale_color_manual(
        values = c("A-" = "#2166AC", "A+" = "#B2182B", "Unknown" = "gray50"),
        name = "Amyloid Status"
      ) +
      labs(
        title = paste(biomarker, "- Aligned to Threshold"),
        subtitle = paste0(title_suffix, " (n=", sila_output$n_subjects, ")"),
        x = "Estimated Time to Threshold (years)",
        y = biomarker
      ) +
      theme_minimal() +
      theme(legend.position = "bottom")
  } else {
    p3 <- ggplot(resfit, aes(x = estdtt0, y = val, group = subid)) +
      geom_line(alpha = 0.3, color = "gray50") +
      geom_point(alpha = 0.3, size = 0.5) +
      geom_line(data = res$tsila, aes(x = adtime, y = val, group = NULL),
                color = "#B2182B", linewidth = 1.2) +
      geom_hline(yintercept = threshold, linetype = "dashed", color = "red") +
      labs(
        title = paste(biomarker, "- Aligned to Threshold"),
        subtitle = paste0(title_suffix, " (n=", sila_output$n_subjects, ")"),
        x = "Estimated Time to Threshold (years)",
        y = biomarker
      ) +
      theme_minimal()
  }

  list(drs = p1, sila = p2, aligned = p3)
}

# Function to extract SILA summary statistics
extract_sila_summary <- function(sila_output) {

  if (is.null(sila_output)) {
    return(tibble(
      biomarker = NA, threshold = NA, n_subjects = NA, n_obs = NA,
      mean_time_to_threshold = NA, sd_time_to_threshold = NA,
      median_time_to_threshold = NA, pct_past_threshold = NA
    ))
  }

  resfit <- sila_output$sila_fit

  # Get baseline (first observation) estimates per subject
  baseline_estimates <- resfit %>%
    group_by(subid) %>%
    slice_min(age, n = 1) %>%
    ungroup()

  tibble(
    biomarker = sila_output$biomarker,
    threshold = sila_output$threshold,
    n_subjects = sila_output$n_subjects,
    n_obs = sila_output$n_obs,
    mean_time_to_threshold = mean(baseline_estimates$estdtt0, na.rm = TRUE),
    sd_time_to_threshold = sd(baseline_estimates$estdtt0, na.rm = TRUE),
    median_time_to_threshold = median(baseline_estimates$estdtt0, na.rm = TRUE),
    pct_past_threshold = mean(baseline_estimates$estdtt0 < 0, na.rm = TRUE) * 100
  )
}
